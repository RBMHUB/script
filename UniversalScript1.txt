-- 拖动 + 关闭功能
local dragging = false
local dragStart, startPos

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
    end
end)

btnTeleport.MouseButton1Click:Connect(function()

    btnTeleport.BackgroundColor3 = states.teleport and Color3.fromRGB(34, 197, 94) or Color3.fromRGB(239, 68, 68)
    btnTeleport.Text = states.teleport and "开启点击传送" or "关闭点击传送"

btnJump.MouseButton1Click:Connect(function()

    btnJump.Text = states.jump and "开启无限跳跃" or "关闭无限跳跃"
end)

btnNoclip.MouseButton1Click:Connect(function()
    states.noclip = not states.noclip
    btnNoclip.BackgroundColor3 = states.noclip and Color3.fromRGB(34, 197, 94) or Color3.fromRGB(239, 68, 68)
    btnNoclip.Text = states.noclip and "开启模型穿越" or "关闭模型穿越"
    
    if states.noclip then
        noclipConnection = RunService.Stepped:Connect(function()
            if character then
                for _, part in ipairs(character:GetDescendants()) do
                    if part:IsA("BasePart") then part.CanCollide = false end
                end
            end
        end)
    else
        if noclipConnection then noclipConnection:Disconnect() end
        if character then
            for _, part in ipairs(character:GetDescendants()) do
                if part:IsA("BasePart") then part.CanCollide = true end
            end
        end
    end
end)

speedConfirm.MouseButton1Click:Connect(function()

        if num then
            defaultWalkSpeed = math.clamp(num, 0, 500)
            humanoid.WalkSpeed = defaultWalkSpeed
            speedConfirm.Text = "✓"
            task.wait(0.8)

end)

mouse.Button1Down:Connect(function()

        local hit = mouse.Hit
        if hit then rootPart.CFrame = CFrame.new(hit.Position + Vector3.new(0, 5, 0)) end
    end
end)
