--// 【现代化坐标传送脚本】 - 缩小版UI + 单行坐标显示
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local player = Players.LocalPlayer
local mouse = player:GetMouse()

local savedPositions = {}
local character, rootPart
local clickSaveEnabled = false
local clickConnection = nil

-- 角色加载
local function setupCharacter(char)
    character = char
    rootPart = char:WaitForChild("HumanoidRootPart")
end
player.CharacterAdded:Connect(setupCharacter)
if player.Character then setupCharacter(player.Character) end

-- 创建UI（缩小约1/3）
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "PositionSaverGui"
screenGui.ResetOnSpawn = false
screenGui.IgnoreGuiInset = true
screenGui.DisplayOrder = 999999
screenGui.Parent = player:WaitForChild("PlayerGui")

local mainFrame = Instance.new("Frame")
mainFrame.Size = UDim2.new(0, 215, 0, 355)        -- 缩小约1/3
mainFrame.Position = UDim2.new(0.02, 0, 0.5, -177)
mainFrame.BackgroundColor3 = Color3.fromRGB(28, 28, 28)
mainFrame.BorderSizePixel = 0
mainFrame.BackgroundTransparency = 0.05
mainFrame.Parent = screenGui

local mainCorner = Instance.new("UICorner")
mainCorner.CornerRadius = UDim.new(0, 14)
mainCorner.Parent = mainFrame

local mainStroke = Instance.new("UIStroke")
mainStroke.Color = Color3.fromRGB(65, 65, 65)
mainStroke.Thickness = 1.5
mainStroke.Parent = mainFrame

-- ==================== 标题栏 ====================
local titleBar = Instance.new("Frame")
titleBar.Size = UDim2.new(1, 0, 0, 36)
titleBar.BackgroundColor3 = Color3.fromRGB(22, 22, 22)
titleBar.Parent = mainFrame

local titleBarCorner = Instance.new("UICorner")
titleBarCorner.CornerRadius = UDim.new(0, 14)
titleBarCorner.Parent = titleBar

-- 标题
local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, -48, 1, 0)
title.BackgroundTransparency = 1
title.Text = "Roblox黑科技"
title.TextColor3 = Color3.fromRGB(240, 240, 240)
title.TextScaled = true
title.Font = Enum.Font.GothamBold
title.TextXAlignment = Enum.TextXAlignment.Left
title.Position = UDim2.new(0, 14, 0, 0)
title.Parent = titleBar

-- 关闭按钮（深灰风格）
local closeBtn = Instance.new("TextButton")
closeBtn.Size = UDim2.new(0, 28, 0, 28)
closeBtn.Position = UDim2.new(1, -34, 0.5, -14)
closeBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
closeBtn.BackgroundTransparency = 0.3
closeBtn.Text = "x"
closeBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
closeBtn.TextScaled = true
closeBtn.Font = Enum.Font.GothamBold
closeBtn.Parent = titleBar

local closeCorner = Instance.new("UICorner")
closeCorner.CornerRadius = UDim.new(0, 9)
closeCorner.Parent = closeBtn

closeBtn.MouseEnter:Connect(function()
    closeBtn.BackgroundTransparency = 0.1
    closeBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
end)
closeBtn.MouseLeave:Connect(function()
    closeBtn.BackgroundTransparency = 0.3
    closeBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
end)

-- 当前坐标（强制单行）
local coordFrame = Instance.new("Frame")
coordFrame.Size = UDim2.new(0.95, 0, 0, 42)
coordFrame.Position = UDim2.new(0.025, 0, 0, 42)
coordFrame.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
coordFrame.Parent = mainFrame

Instance.new("UICorner", coordFrame).CornerRadius = UDim.new(0, 10)

local coordLabel = Instance.new("TextLabel")
coordLabel.Size = UDim2.new(1, 0, 1, 0)
coordLabel.BackgroundTransparency = 1
coordLabel.Text = "X: 0.00  Y: 0.00  Z: 0.00"
coordLabel.TextColor3 = Color3.fromRGB(180, 255, 180)
coordLabel.TextScaled = true
coordLabel.Font = Enum.Font.GothamSemibold
coordLabel.TextXAlignment = Enum.TextXAlignment.Left
coordLabel.TextTruncate = Enum.TextTruncate.AtEnd   -- 防止换行，过长自动省略
coordLabel.Parent = coordFrame

-- 按钮区域（缩小）
local saveBtn = Instance.new("TextButton")
saveBtn.Size = UDim2.new(0.46, 0, 0, 34)
saveBtn.Position = UDim2.new(0.025, 0, 0, 92)
saveBtn.BackgroundColor3 = Color3.fromRGB(0, 175, 105)
saveBtn.Text = "保存当前位置"
saveBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
saveBtn.TextScaled = true
saveBtn.Font = Enum.Font.GothamBold
saveBtn.Parent = mainFrame
Instance.new("UICorner", saveBtn).CornerRadius = UDim.new(0, 9)

local clickSaveBtn = Instance.new("TextButton")
clickSaveBtn.Size = UDim2.new(0.46, 0, 0, 34)
clickSaveBtn.Position = UDim2.new(0.525, 0, 0, 92)
clickSaveBtn.BackgroundColor3 = Color3.fromRGB(190, 55, 55)
clickSaveBtn.Text = "保存点击坐标"
clickSaveBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
clickSaveBtn.TextScaled = true
clickSaveBtn.Font = Enum.Font.GothamBold
clickSaveBtn.Parent = mainFrame
Instance.new("UICorner", clickSaveBtn).CornerRadius = UDim.new(0, 9)

local clearBtn = Instance.new("TextButton")
clearBtn.Size = UDim2.new(0.95, 0, 0, 32)
clearBtn.Position = UDim2.new(0.025, 0, 0, 134)
clearBtn.BackgroundColor3 = Color3.fromRGB(190, 55, 55)
clearBtn.Text = "清空所有坐标"
clearBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
clearBtn.TextScaled = true
clearBtn.Font = Enum.Font.GothamBold
clearBtn.Parent = mainFrame
Instance.new("UICorner", clearBtn).CornerRadius = UDim.new(0, 9)

-- 列表标题
local listTitle = Instance.new("TextLabel")
listTitle.Size = UDim2.new(0.95, 0, 0, 20)
listTitle.Position = UDim2.new(0.025, 0, 0, 175)
listTitle.BackgroundTransparency = 1
listTitle.Text = "点击传送 (0 / 99)"
listTitle.TextColor3 = Color3.fromRGB(200, 200, 200)
listTitle.TextScaled = true
listTitle.Font = Enum.Font.Gotham
listTitle.TextXAlignment = Enum.TextXAlignment.Left
listTitle.Parent = mainFrame

-- 滚动框
local scroll = Instance.new("ScrollingFrame")
scroll.Size = UDim2.new(0.95, 0, 0, 155)
scroll.Position = UDim2.new(0.025, 0, 0, 200)
scroll.BackgroundTransparency = 1
scroll.ScrollBarThickness = 5
scroll.Parent = mainFrame

local listLayout = Instance.new("UIListLayout")
listLayout.Padding = UDim.new(0, 5)
listLayout.Parent = scroll

-- 实时坐标更新
RunService.Heartbeat:Connect(function()
    if rootPart then
        local pos = rootPart.Position
        coordLabel.Text = string.format("X: %.2f  Y: %.2f  Z: %.2f", pos.X, pos.Y, pos.Z)
    end
end)

-- 保存当前位置
saveBtn.MouseButton1Click:Connect(function()
    if #savedPositions >= 99 or not rootPart then return end
    local pos = rootPart.Position
    table.insert(savedPositions, pos)
    CreateListItem(pos)
end)

-- 点击保存坐标
clickSaveBtn.MouseButton1Click:Connect(function()
    clickSaveEnabled = not clickSaveEnabled
    if clickSaveEnabled then
        clickSaveBtn.BackgroundColor3 = Color3.fromRGB(0, 175, 105)
        clickSaveBtn.Text = "停止点击保存"
        clickConnection = mouse.Button1Down:Connect(function()
            if not clickSaveEnabled or #savedPositions >= 99 then return end
            local hit = mouse.Hit
            if hit then
                local pos = hit.Position
                table.insert(savedPositions, pos)
                CreateListItem(pos)
            end
        end)
    else
        clickSaveBtn.BackgroundColor3 = Color3.fromRGB(190, 55, 55)
        clickSaveBtn.Text = "保存点击坐标"
        if clickConnection then clickConnection:Disconnect() end
    end
end)

-- 清空
clearBtn.MouseButton1Click:Connect(function()
    savedPositions = {}
    for _, v in ipairs(scroll:GetChildren()) do
        if v:IsA("TextButton") then v:Destroy() end
    end
    scroll.CanvasSize = UDim2.new(0,0,0,0)
    listTitle.Text = "点击传送 (0 / 99)"
end)

-- 关闭按钮
closeBtn.MouseButton1Click:Connect(function()
    screenGui:Destroy()
    print("坐标传送器 UI 已关闭")
end)

-- 创建列表项（强制单行显示）
function CreateListItem(pos)
    local item = Instance.new("TextButton")
    item.Size = UDim2.new(1, 0, 0, 32)
    item.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    item.Text = string.format("%d. X: %.2f Y: %.2f Z: %.2f", #savedPositions, pos.X, pos.Y, pos.Z)
    item.TextColor3 = Color3.fromRGB(255, 255, 255)
    item.TextScaled = true
    item.Font = Enum.Font.GothamSemibold
    item.TextXAlignment = Enum.TextXAlignment.Left
    item.TextTruncate = Enum.TextTruncate.AtEnd   -- 关键：防止换行，过长自动截断
    item.Parent = scroll
    
    Instance.new("UICorner", item).CornerRadius = UDim.new(0, 9)

    item.MouseButton1Click:Connect(function()
        if rootPart then
            rootPart.CFrame = CFrame.new(pos + Vector3.new(0, 5, 0))
        end
    end)

    scroll.CanvasSize = UDim2.new(0, 0, 0, listLayout.AbsoluteContentSize.Y)
    listTitle.Text = string.format("点击传送 (%d / 99)", #savedPositions)
end

-- 窗口拖动
local dragging = false
local dragStart, startPos

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
    end
end)

print("✅ 缩小版坐标传送脚本已加载 | UI缩小约1/3 + 坐标强制单行显示")
